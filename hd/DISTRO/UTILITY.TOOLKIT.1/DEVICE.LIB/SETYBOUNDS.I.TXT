PROCEDURE SETYBOUNDS(MIN,MAX:INTEGER);
BEGIN
#A
 STX _T
 LDA MSLOT
 BEQ SYB9
 LDY #5         ;MAX LSB
 LDA (_SP),Y
 STA $4F8
 INY
 LDA (_SP),Y
 STA $5F8
 INY
 LDA (_SP),Y    ;MIN LSB
 STA $478
 INY
 LDA (_SP),Y
 STA $578
 LDY #$17       ;CLAMPMOUSE
 JSR MADR
 LDA #1         ;CHANGE Y LIMITS
 JSR MCALL
 LDY #$18
 JSR MADR
 JSR MCALL
SYB9 LDX _T
#
END;
