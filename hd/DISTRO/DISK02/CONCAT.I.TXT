PROCEDURE CONCAT(VAR A1,A2,A3:STRING);                 
VAR I,J:INTEGER;
BEGIN
 I:=MAXSTRING;
 WHILE ((A1[I] = ' ') AND (I <> 1))
  DO I:=I-1;
 J:=1;
 IF (A1[I] <> ' ') THEN BEGIN
   WHILE (J <= I) DO BEGIN
     A3[J] := A1[J];
     J:=J+1;
   END;
 END;

 I:=1;
 WHILE (J <= MAXSTRING) DO BEGIN
   A3[J] := A2[I];
   I:=I+1;
   J:=J+1;
 END;
END;
