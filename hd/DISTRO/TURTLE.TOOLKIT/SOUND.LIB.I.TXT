PROCEDURE CLICK;
BEGIN
#A
 STX _T
 LDY #5
CK_2 BIT $C030
 LDX #250
CK_1 DEX
 BNE CK_1
 DEY
 BNE CK_2
 LDX _T
#
END;

PROCEDURE BELL;
BEGIN
#A
 STX _T
 LDA #7
 JSR $FDED
 LDX _T
#
END;

PROCEDURE NOTE(PITCH,DURATION:INTEGER);
BEGIN
#A
 STX _T
 LDY #5
 LDA (_SP),Y
 STA _T+2       ;DURATION
 INY
 INY
 LDA (_SP),Y
 STA _T+1       ;PITCH
NOTE1 LDA $C030
NOTE4 DEY
 BNE NOTE2
 DEC _T+2
 BEQ NOTE3
NOTE2 DEX
 BNE NOTE4
 LDX _T+1
 JMP NOTE1
NOTE3 LDX _T
#
END;

PROCEDURE PHASER(DURATION:INTEGER);
VAR X,Y : INTEGER;
BEGIN
   FOR X := 1 TO DURATION DO
   BEGIN
      Y := 25;
      REPEAT
         NOTE(Y,5);
         Y := Y + 10
      UNTIL Y > 100
   END
END;
