PROCEDURE GETTIME(VAR HOURS,MINUTES:INTEGER);
BEGIN
#A
 STX _T
 LDY #5
GTT1 LDA (_SP),Y
 STA _T-4,Y
 INY
 CPY #9
 BNE GTT1
 JSR _MLI
 DB  $82       ;GET_TIME
 DW  0
 LDY #1
 STA (_T+1),Y  ;MSB OF MINUTES
 STA (_T+3),Y  ;AND HOURS IS ZERO
 DEY
 LDA $BF92
 STA (_T+1),Y
 LDA $BF93
 STA (_T+3),Y
 LDX _T
#
END;
