PROGRAM RANDOM_DEMO;

CONST
   MIN=1;
   MAX=50;

VAR
  TARGET,X1,X2,X3,X4:INTEGER;
  CH:CHAR;

#i rnd.i
#i random.i
#i seed.i

PROCEDURE GUESS;
VAR X:INTEGER;
BEGIN
   TARGET:=RANDOM(MIN,MAX);
   REPEAT
      WRITE('ENTER YOUR GUESS? ');
      READLN(X);
      IF X>TARGET THEN WRITELN('TOO HIGH')
                  ELSE
      IF X<TARGET THEN WRITELN('TOO LOW')
                  ELSE
         WRITELN('YOU GOT IT!!!!')
   UNTIL X=TARGET
END;


BEGIN
   WRITELN('*** RANDOM NUMBER DEMO ***');
   WRITELN;
   WRITELN('AS THE GAME PROCEEDS AND YOU');
   WRITELN('GUESS THE NUMBERS, THE SEEDS');
   WRITELN('WILL CHANGE.');
   WRITELN;
   WRITELN('NUMBER ARE BETWEEN ',MIN,' AND ',MAX);
   WRITELN;
   REPEAT
      GUESS;
      WRITELN; WRITE('AGAIN? '); READLN(CH);
      IF CH='Y' THEN
      BEGIN
         WRITELN('TYPE 4 INTEGERS SO THAT THE');
         WRITELN('RANDOM NUMBER SEED WILL CHANGE');
         WRITE('NUMBERS? : ');
         READLN(X1,X2,X3,X4);
         SEED(X2,X3,X1,X4)
     END
   UNTIL CH<>'Y'
END.
