#include <sgtty.h>
#include <device.h>

_isdev(name)
char *name;
{
	register struct _name_dev *dp;

	for (dp=&_dev_info->dev_con;dp<=&_dev_info->dev_ser;dp++) 
		if (strequ(name, dp->dev_nam) == 0) 
			return -1;
	if (name[3] == 0 && name[2] == ':' && toupper(name[0]) == 'S') 
		return -1;
	return 0;
}
