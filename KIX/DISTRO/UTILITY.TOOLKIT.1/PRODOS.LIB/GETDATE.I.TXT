PROCEDURE GETDATE(VAR MONTH,DAY,YEAR:INTEGER);
BEGIN
#A
 STX _T
 LDY #5
GDT1 LDA (_SP),Y
 STA _T-4,Y
 INY
 CPY #11
 BNE GDT1
 JSR _MLI
 DB  $82       ;GET_TIME
 DW  0
 LDY #1
 STA (_T+1),Y   ;YEAR MSB
 STA (_T+3),Y   ;DAY MSB
 STA (_T+5),Y   ;MONTH MSB
 DEY
 LDA $BF91
 LSR
 STA (_T+1),Y
 LDA $BF90
 PHA
 ROR
 LSR
 LSR
 LSR
 LSR
 STA (_T+5),Y
 PLA
 AND #$1F
 STA (_T+3),Y
 LDX _T
#
END;
