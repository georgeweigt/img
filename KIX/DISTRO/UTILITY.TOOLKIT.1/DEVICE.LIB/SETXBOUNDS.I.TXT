PROCEDURE SETXBOUNDS(MIN,MAX:INTEGER);
BEGIN
#A
 STX _T
 LDX MSLOT
 BEQ SXB9
 LDY #5         ;MAX LSB
 LDA (_SP),Y
 STA $4F8
 INY
 LDA (_SP),Y
 STA $5F8
 INY
 LDA (_SP),Y    ;MIN LSB
 STA $478
 INY
 LDA (_SP),Y
 STA $578
 LDY #$17       ;CLAMPMOUSE
 JSR MADR
 LDA #0         ;CHANGE X LIMITS
 JSR MCALL
 LDY #$18       ;HOMEMOUSE
 JSR MADR
 JSR MCALL
SXB9 LDX _T
#
END;
